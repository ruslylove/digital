\documentclass[border=10pt]{standalone}
\usepackage{circuitikz}

\ctikzset{
logic ports=ieee,
logic ports/scale=0.7,
}

\begin{document}
\begin{circuitikz}[]

    % Gate 1: The standard NOR operation
    \draw (0,0) node[nor port] (NOR1) {};
    
    % Gate 2: The Inverter (NOR configured as NOT)
    \draw (2,0) node[nor port] (INV) {};

    % Inputs A and B to the first gate
    \draw (NOR1.in 1) -- ++(-1,0) node[left] {$A$};
    \draw (NOR1.in 2) -- ++(-1,0) node[left] {$B$};

    % Connect output of first gate to BOTH inputs of the second gate
    \draw (NOR1.out) |- coordinate[pos=0.0](split) (INV.in 1);
    \draw (split) node[circ]{} |- (INV.in 2);

    % Output
    \draw (INV.out) -- ++(1,0) node[right] {$Y = A + B$};

\end{circuitikz}
\end{document}
