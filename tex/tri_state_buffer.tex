\documentclass[border=10pt]{standalone}
\usepackage{circuitikz}
\ctikzset{logic ports=ieee}

\begin{document}
\begin{circuitikz}
    % Buffer port
    \node[buffer port] (buf) at (0,0) {};
    
    % Input
    \draw (buf.in 1) -- ++(-1,0) node[left] {$Input$};
    
    % Output
    \draw (buf.out) -- ++(1,0) node[right] {$Output$};
    
    % Control (Enable) - connecting to the top center of the logic gate body
    % buffer port (IEEE) has an 'up' anchor for this purpose.
    \draw (buf.up) -- ++(0, 0.5) node[above] {$Control (C)$};

    % High-Impedance notation (optional, but requested symbol)
    \node[below=0.5cm] at (buf.south) {Tri-State Buffer};
\end{circuitikz}
\end{document}
