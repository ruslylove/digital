\documentclass[border=10pt]{standalone}
\usepackage{circuitikz}
\usetikzlibrary{calc}

\begin{document}
\begin{circuitikz}[american]
    \ctikzset{logic ports=ieee}
    % Draw two NAND gates
    \draw (0,2) node[nand port] (nand1) {};
    \draw (0,0) node[nand port] (nand2) {};

    % Inputs (Active Low S' and R')
    % Usually Top is S', Bottom is R' for standard Q/Q' alignment
    \draw (nand1.in 1) -- ++(-1,0) node[left] {$\overline{S}$};
    \draw (nand2.in 2) -- ++(-1,0) node[left] {$\overline{R}$};

    % Outputs
    \draw (nand1.out) -- ++(1,0) node[right] {$Q$};
    \draw (nand2.out) -- ++(1,0) node[right] {$Q'$};

    % Feedback connections (Cross-coupling)
    % From Top Output to Bottom Input 1
    \draw (nand1.out) -- ++(0.2,0) coordinate(feedQ) -- ++(0,-0.5) -- ($(nand2.in 1) + (0,0.5)$) -- (nand2.in 1);
    
    % From Bottom Output to Top Input 2
    \draw (nand2.out) -- ++(0.2,0) coordinate(feedQbar) -- ++(0,0.5) -- ($(nand1.in 2) + (0,-0.5)$) -- (nand1.in 2);

    % Add nodes
    \node[circ] at (feedQ) {};
    \node[circ] at (feedQbar) {};

\end{circuitikz}
\end{document}
