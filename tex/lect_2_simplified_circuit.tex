\documentclass[border=10pt]{standalone}
\usepackage{circuitikz}

\ctikzset{
    logic ports=ieee,
    logic ports/scale=1,
}

\begin{document}
\begin{circuitikz}[thick]
    % OR gate
    \node[or port] (OR) at (3, 0) {};
    
    % Input x1 with inverter
    \node[not port, scale=0.5] (NOT) at (0.5, 0.28) {};
    \draw (NOT.out) -- (OR.in 1);
    \draw (NOT.in) -- ++(-0.5,0) node[left] {$x_1$};
    
    % Input x2
    \draw (OR.in 2) -- ++(-2.4,0) node[left] {$x_2$};
    
    % Output
    \draw (OR.out) -- ++(0.5,0) node[right] {$f$};
\end{circuitikz}
\end{document}
