\usepackage[american]{circuitikz}

\begin{circuitikz}
    \ctikzset{logic ports=ieee} % Use IEEE standard symbols

    % --- Inputs ---
   % \node (A) at (0, 4) {A};
   % \node (B) at (0, 3) {B};
   % \node (Cin) at (0, 1) {$C_{in}$};

    % --- First Half-Adder ---
    % XOR for the intermediate sum
    \node[xor port] (XOR1) at (2, 3.5) {};
    % AND for the first carry
    \node[and port] (AND1) at (2, 1) {};

    % --- Second Half-Adder ---
    % XOR for the final Sum
    \node[xor port] (XOR2) at (5.5, 2.5) {};
    % AND for the second carry
    \node[and port] (AND2) at (5.5, 0) {};
    

    % --- Final OR for Carry Out ---
    \node[or port, anchor=in 2] (OR1) at (7.5, 0) {};

    % connect or1 in2 to and2 out
    

    % --- Connections ---
    % Inputs to first HA
 

    \draw (XOR1.in 1) -- ++(-1.5,0) node[left] {$A$} coordinate[pos=0.2] (pointA);
    \draw (XOR1.in 2) -- ++(-1.5,0) node[left] {$B$} coordinate[pos=0.6] (pointB);
    \draw (pointA) to[short, *-] (pointA |- AND1.in 1) -- (AND1.in 1);
    \draw (pointB) to[short, *-] (pointB |- AND1.in 2) -- (AND1.in 2);

    % Inputs to second HA
    \draw (XOR2.in 2) -- ++(-5,0) node[left] {$C_{in}$} coordinate[pos=0.1] (pointCin);
    \draw (pointCin) to[short, *-] (pointCin |- AND2.in 1) -- (AND2.in 1);

    % First HA to Second HA
    \draw (XOR1.out) -| (XOR2.in 1) coordinate[pos=0.1] (pointXOR1out);
    \draw (pointXOR1out) to[short, *-] (pointXOR1out |- AND2.in 2) -- (AND2.in 2);

    % Carry connections to final OR gate
   % \draw (AND1.out) |- (OR1.in 1) {};
  %  \draw (AND1.out) -- ++(3.5,0) |- (OR1.in 1);
    \draw (AND1.out) -| (OR1.in 1);
    \draw (AND2.out) -- (OR1.in 2);
  

    % --- Outputs ---
    \draw (XOR2.out) -- ++(3.5,0) node[right] {Sum ($S$)};
    \draw (OR1.out) -- ++(0.5,0) node[right] {Carry Out ($C_{out}$)};

\end{circuitikz}
