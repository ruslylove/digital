\documentclass[border=10pt]{standalone}
\usepackage[american]{circuitikz}

% Define a new component style for the Half-Adder (HA) block
\tikzset{
    halfadder/.style={
        draw,
        rectangle,
        minimum width=1.5cm,
        minimum height=2cm,
        label=center:HA, 
        anchor=center, % This is important for positioning
        append after command={
            \pgfextra{\let\mainnode=\tikzlastnode}
            node at (\mainnode.north west) [yshift=-0.4cm] (A){}
            node at (\mainnode.south west) [yshift=+0.4cm] (B){}
            node at (\mainnode.north east) [yshift=-0.4cm] (S){}
            node at (\mainnode.south east) [yshift=0.4cm] (C){}
        }
    }
}

\begin{document}
\begin{circuitikz}[]
    \ctikzset{logic ports=ieee} % Use IEEE standard symbols

    % --- Place the components ---
    % First Half-Adder
    \node[halfadder] (HA3) at (0, 0.5) {};
    \node at (HA3.west) [yshift=+0.4cm] (sA){};
    \node at (HA3.west) [yshift=-0.4cm] (sB){};
    \node at (HA3.east) [yshift=+0.4cm] (sS){};
    \node at (HA3.east) [yshift=-0.4cm] (sC){};



    % Second Half-Adder
    \node[halfadder] (HA2) at (3, -0.5) {};
    % OR gate for the final carry
    \node[or port, anchor=in 1, scale=1] (OR1) at (5, -1.1) {};

    % --- Connect inputs ---
    \draw (HA3.west |- sA) -- ++(-1,0) node[left] {$A$};
    \draw (HA3.west |- sB) -- ++(-1,0) node[left] {$B$};

    % --- Connect HA1 to HA2 ---
    % Sum of HA1 goes to input A of HA2
   % \draw (HA1.north east |- S) -- (HA2.west |- A);
   % \draw (HA1.east |- C) -- ++(0.5,0) coordinate (pointB);
    \draw (HA3.east |- sS) -- +(1.2,0) |- (HA2.west |- A);

    % --- Connect carry logic ---
    % Carry of HA1 goes to the OR gate
    %\draw (pointB) -- (pointB |- OR1.in 2) |- (OR1.in 2);
    % Carry of HA2 goes to the OR gate
    \draw (HA2.east |- C) -- (OR1.in 1);

    \draw (HA3.east |- sC) -- +(0.5,0) |- (OR1.in 2);

    % --- Connect outputs ---
    \draw (HA2.east |- S) -- ++(4,0) node[right] {Sum ($S$)};
    \draw (OR1.out) -- ++(0.6,0) node[right] {Carry Out ($C_{out}$)};

    \draw (HA2.west |- B) -- ++(-0.25,0) -- ++(-3.8,0) node[left] {$C_{in}$};





\end{circuitikz}
\end{document}
