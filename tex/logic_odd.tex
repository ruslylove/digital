\documentclass[border=10pt]{standalone}
% We explicitly load the 'ieee' style option here.
% This ensures strictly IEEE standard logic gates (Curved/Distinctive).
\usepackage{circuitikz}

\ctikzset{
logic ports=ieee,
logic ports/scale=0.7,
}

\begin{document}

\begin{circuitikz}[]

    % ==========================
    % 1. XOR GATE (IEEE Shape)
    % ==========================
    % The 'xor port' automatically uses the IEEE distinctive shape
    % because of the package option above.
    \draw (0,0) node[xor port, number inputs=3] (XOR) {};
    
    % Inputs
    \draw (XOR.in 1) -- ++(-1,0) node[left] {$A$};
    \draw (XOR.in 2) -- ++(-1,0) node[left] {$B$};
    \draw (XOR.in 3) -- ++(-1,0) node[left] {$C$};
    
    % Output
    \draw (XOR.out) -- ++(1,0) node[right] {$Y = A \oplus B \oplus C$};

    % Label
    \node at (XOR.down) [below=0.3cm] {};


\end{circuitikz}

\end{document}
