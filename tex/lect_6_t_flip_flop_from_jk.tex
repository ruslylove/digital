\documentclass[border=10pt]{standalone}
\usepackage{circuitikz}

\begin{document}
\begin{circuitikz}[american]
    \ctikzset{logic ports=ieee}
    
    % JK Flip-Flop
    \node[flipflop JK] (JK) at (0,0) {};
    
    % T Input Wiring
    % T aligned with J (pin 1) and connected to J and K (pin 3)
    \draw (JK.pin 1) -- ++(-1.5,0) node[left] {$T$};
    \draw (JK.pin 3) -- ++(-0.5,0) coordinate (k_in);
    
    % Vertical connection from T line (near J) down to K
    \draw (JK.pin 1) ++(-0.5,0) coordinate (j_conn);
    \draw (j_conn) -- (k_in);
    \node[circ] at (j_conn) {}; % Connection dot
    
    % Clock (pin 2)
    \draw (JK.pin 2) -- ++(-1,0) node[left] {CLK};
    
    % Outputs
    \draw (JK.pin 6) -- ++(0.5,0) node[right] {$Q$};
    \draw (JK.pin 4) -- ++(0.5,0) node[right] {$\bar{Q}$};

\end{circuitikz}
\end{document}
