\documentclass[dvisvgm]{standalone}
\usepackage[european]{circuitikz}

\begin{document}
\begin{circuitikz}
    % Inputs
    \node (M) at (-2, 2) {$M$};
    \node (D) at (-2, 0) {$D$};
    \node (V) at (-2, -1) {$V$};
    \node (Q) at (-2, 1) {$Q$};

    % Gates
    % OR gate for Q + D + V
    % Since standard or gate has fewer inputs, we'll cascade or use a larger one conceptually
    % Let's use a 3-input OR
    \node[or port, number inputs=3] (OR1) at (1, 0) {};
    
    % AND gate for M * (OR1)
    \node[and port] (AND1) at (3, 1) {};

    % Flip Flop
    \node[flipflop D] (FF) at (5, 1) {};
    \node[right] at (FF.bup) {CLK};
    \node (CLK) at (3.5, -0.5) {CLK};
    
    % Connections
    \draw (Q) -- (OR1.in 1);
    \draw (D) -- (OR1.in 2);
    \draw (V) -- (OR1.in 3);
    
    \draw (M) |- (AND1.in 1);
    \draw (OR1.out) |- (AND1.in 2);
    
    \draw (AND1.out) -- (FF.pin 1); % D input
    \draw (CLK) -| (FF.pin 3);
    
    % Output loopback Q
    \draw (FF.pin 6) -- ++(1,0) node[right] (Output) {$A (Alarm)$}
          (FF.pin 6) -- ++(0.5,0) -- ++(0, 2) -- (-2.5, 3) |- (Q);

\end{circuitikz}
\end{document}
