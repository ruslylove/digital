\documentclass[border=10pt]{standalone}
% We explicitly load the 'ieee' style option here.
% This ensures strictly IEEE standard logic gates (Curved/Distinctive).
\usepackage{circuitikz}

\ctikzset{
logic ports=ieee,
logic ports/scale=1,
}

\begin{document}

\begin{circuitikz}[]

    % ==========================
    % 1. XNOR GATE (IEEE Shape)
    % ==========================
    % The 'xnor port' automatically uses the IEEE distinctive shape
    % because of the package option above.
    \draw (0,0) node[xnor port, number inputs=3] (XNOR) {};
    
    % Inputs
    \draw (XNOR.in 1) -- ++(-1,0) node[left] {$A$};
    \draw (XNOR.in 2) -- ++(-1,0) node[left] {$B$};
    \draw (XNOR.in 3) -- ++(-1,0) node[left] {$C$};
    
    % Output
    \draw (XNOR.out) -- ++(1,0) node[right] {$Y = \overline{A \oplus B \oplus C}$};

    % Label
    \node at (XNOR.down) [below=0.3cm] {};

\end{circuitikz}

\end{document}
