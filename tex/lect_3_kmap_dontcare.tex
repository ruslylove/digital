\documentclass[border=10pt]{standalone}
\usepackage{karnaugh-map}

\begin{document}

% 4 Variables: 4x4 grid
% Rows = wx, Columns = yz
\begin{karnaugh-map}[4][4][1][$yz$][$wx$]

    % 1. Place Minterms (The '1's)
    \minterms{1, 3, 7, 11, 15}

    % 2. Place Don't Cares (The 'X's)
    % The command \indeterminants places an 'X' in the specified cells.
    % These can be treated as 1 or 0 to help form larger groups.
    \terms{0, 2, 5}{X}

    % 3. Grouping (Simplification)
    
    % Group 1: Vertical Column (yz)
    % Covers 3, 7, 15, 11.
    % Simplifies to: yz
    \implicant{3}{11}

    % Group 2: Top Row (w'x')
    % We use Don't Cares at 0 and 2 to group 0, 1, 3, 2.
    % Since 3 is already covered, we still include it to make the group larger (size 4).
    % Simplifies to: w'x'
    \implicant{0}{2}

\end{karnaugh-map}

\end{document}
